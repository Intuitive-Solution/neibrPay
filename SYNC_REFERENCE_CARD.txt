⚙️ SYNC STRATEGY
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

    INITIAL SYNC (First time)
    └─ Lookback:    730 days (2 years)
    └─ Purpose:     Complete historical baseline
    └─ Detection:   sync_start_date = NULL
    └─ Frequency:   Once per account
    └─ Time:        2-10 seconds

    ONGOING SYNC (Recurring)
    └─ Lookback:    30 days (4-5 weeks)
    └─ Purpose:     Catch pending→posted transitions
    └─ Detection:   sync_start_date = SET
    └─ Frequency:   Every Monday 00:00 UTC
    └─ Time:        <3 seconds per account


🗓️ SYNC SCHEDULE
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

    AUTOMATED SYNC
    Every Monday at 00:00 UTC
    └─ Triggered by: Laravel Kernel scheduler
    └─ Job: SyncPlaidTransactions
    └─ Command: plaid:sync
    └─ Window: Previous 30 days

    MANUAL SYNC (On Demand)
    "Refresh Transactions Now" button (Settings → Bank)
    └─ Syncs immediately (all accounts)
    └─ Window: Previous 30 days
    └─ Shows success/error message

    UI REFRESH (Display Only)
    "Refresh" button (Transactions page)
    └─ Reloads from database (NO API call)
    └─ Fast operation (<1 second)
    └─ Updates transaction display

💡 WHY THESE VALUES?
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

    730 DAYS (Initial)
    └─ Maximum Plaid allows
    └─ Complete 2-year history for reconciliation
    └─ Ensures baseline completeness

    30 DAYS (Ongoing)
    └─ Covers pending→posted window (1-3 days)
    └─ Handles delayed postings (up to 5 days)
    └─ Includes overlap for transaction updates
    └─ Efficient with weekly syncs

    WEEKLY/MONDAY
    └─ Sufficient for ~4 transactions/month
    └─ Predictable schedule for users
    └─ Off-peak timing (Monday 00:00 UTC)
    └─ Industry standard practice
